🎨 SCSS (style.scss)
:root {
  --bg-color: white;
  --text-color: black;
}

body.dark {
  --bg-color: #111;
  --text-color: #eee;
}

body {
  background: var(--bg-color);
  color: var(--text-color);
  transition: 0.3s ease;
}

⚙️ JavaScript (script.js)
const btn = document.getElementById("toggleBtn");

btn.addEventListener("click", () => {
  document.body.classList.toggle("dark");
});
